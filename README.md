    function formatUSD(n){
      return '$' + Number(n).toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    }
    function formatCompactUSD(n){
      const abs = Math.abs(n);
      if (abs >= 1e12) return '$' + (n / 1e12).toFixed(2) + 'T';
      if (abs >= 1e9)  return '$' + (n / 1e9).toFixed(2) + 'B';
      if (abs >= 1e6)  return '$' + (n / 1e6).toFixed(2) + 'M';
      if (abs >= 1e3)  return '$' + (n / 1e3).toFixed(2) + 'K';
      return formatUSD(n);
    }

    function createSparklinePath(values, w=320, h=54, pad=6){
      if (!values || values.length < 2) return "";
      const min = Math.min(...values);
      const max = Math.max(...values);
      const range = (max - min) || 1;
      const innerW = w - pad*2;
      const innerH = h - pad*2;

      return values.map((v,i) => {
        const x = pad + (i/(values.length-1)) * innerW;
        const y = pad + (1 - (v - min)/range) * innerH;
        return `${i===0 ? "M" : "L"} ${x.toFixed(2)} ${y.toFixed(2)}`;
      }).join(" ");
    }

    function renderGallerySkeleton(){
      const grid = $("#galleryGrid");

    async function updateGallery(){
      try{
        const ids = COINS.map(x => x.id).join(",");
        const url =
          "https://api.coingecko.com/api/v3/coins/markets" +
          "?vs_currency=usd" +
          `&ids=${encodeURIComponent(ids)}` +
          "&sparkline=true" +
          "&price_change_percentage=24h";

        const res = await fetch(url, { cache:"no-store" });
        if(!res.ok) throw new Error("HTTP " + res.status);
        const list = await res.json();

        list.forEach(coin => {
          const card = document.querySelector(`.gcard[data-coin="${coin.id}"]`);
          if(!card) return;

          const img = card.querySelector("[data-img]");
          const nameEl = card.querySelector("[data-name]");
          const priceEl = card.querySelector("[data-price]");
          const changeEl = card.querySelector("[data-change]");
          const volEl = card.querySelector("[data-vol]");
          const mcapEl = card.querySelector("[data-mcap]");
          const pathEl = card.querySelector("[data-path]");

          if(img) img.src = coin.image || "";
          if(nameEl) nameEl.textContent = coin.name || coin.id;
          if(priceEl) priceEl.textContent = formatUSD(coin.current_price);

          const chg = coin.price_change_percentage_24h;
          if(changeEl && typeof chg === "number"){
            const pos = chg >= 0;
            changeEl.textContent = (pos ? "+" : "") + chg.toFixed(2) + "% (24H)";
            changeEl.className = "c " + (pos ? "pos" : "neg");
          }

          if(volEl) volEl.textContent = formatCompactUSD(coin.total_volume);
          if(mcapEl) mcapEl.textContent = formatCompactUSD(coin.market_cap);

          const vals = coin?.sparkline_in_7d?.price || [];
          const d = createSparklinePath(vals);
          if(pathEl) pathEl.setAttribute("d", d);
        });

      }catch(e){
        console.error("Gallery update failed:", e);
      }
    }

  // ===== MOBILE MENU (LEBIH RAPiH) =====
  (function(){
    const btn = $("#mobileMenuBtn");
    const nav = $(".nav-links");
    if(!btn || !nav) return;

    function openMenu(){
      nav.classList.add("active");
      btn.innerHTML = '<i class="fas fa-times"></i>';
      document.body.style.overflow = "hidden"; // kunci scroll
    }

    function closeMenu(){
      nav.classList.remove("active");
      btn.innerHTML = '<i class="fas fa-bars"></i>';
      document.body.style.overflow = ""; // balikin scroll
    }

    btn.addEventListener("click", () => {
      nav.classList.contains("active") ? closeMenu() : openMenu();
    });

    // klik link = tutup
    $all(".nav-links a").forEach(a => a.addEventListener("click", closeMenu));

    // klik area luar menu = tutup
    document.addEventListener("click", (e) => {
      if(!nav.classList.contains("active")) return;
      const clickInsideNav = nav.contains(e.target);
      const clickOnBtn = btn.contains(e.target);
      if(!clickInsideNav && !clickOnBtn) closeMenu();
    });

    // tekan ESC = tutup (buat desktop juga)
    document.addEventListener("keydown", (e) => {
      if(e.key === "Escape") closeMenu();
    });

    // jika layar dibesarkan ke desktop, pastikan normal
    window.addEventListener("resize", () => {
      if(window.innerWidth > 768){
        closeMenu();
      }
    });
  })();
</script>
